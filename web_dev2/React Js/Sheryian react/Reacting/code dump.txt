import React, { useState } from "react";

function App() {
  // Use state only can be used inside function
  const [val, setVal] = useState([
    { name: "vinit", age: 20, gender: "male" },
    { name: "Akash", age: 30, gender: "male" },
    { name: "Vipul", age: 26, gender: "male" },

  ]);
  // function OnClickFunc() {
  //   setVal(prev => prev+1);
  // }
  return (
    <div className="m-10">
      {val.map((item, index) => (
        <div key={index}>
          <h1>{item.name}</h1>
          <h1>{item.age}</h1>
          <h1>{item.gender}</h1>
        </div>
      ))}
      <button
        onClick={() =>
          setVal(() =>
            val.map((item) =>
              item.name === "vinit" ? {name:item.name, age: item.age+1,gender:item.gender } : item
            ),
          )
        }
        className={`px-2 py-1 rounded-full text-xs ${val.age >= 18 ? "bg-blue-500" : "bg-red-500"} text-white`}
      >
        Change value
      </button>
    </div>

    // <div className="m-10">
    //   <h1>Name: {val.name}</h1>
    //   <h1>Age: {val.age}</h1>
    //   <h1>isValid: {val.isValid }</h1>
    //   <button onClick={() => setVal({ ...val, age:val.age+1, isValid:(val.age>=18?"true":"false")})} className={`px-2 py-1 rounded-full text-xs ${val.age>=18 ?"bg-blue-500":"bg-red-500"} text-white`}>Change value</button>
    // </div>
  );
}

export default App;
